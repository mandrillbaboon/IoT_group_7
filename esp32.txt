#include <WiFi.h>
#include <PubSubClient.h>
#include "DHT.h"
#include <ESP32Servo.h>
#include <Keypad.h>

// --- CONFIGURATION ---
#define DHTPIN 25      // Pin de ton capteur
#define DHTTYPE DHT22
#define SENSOR_ID 6
#define SERVO_PIN 26   // Pin de ton éolienne
#define MQTT_TOPIC "esp8266/dht11"

const char* ssid = "Poznan Dream";
const char* password = "Pipapopu";
const char* mqtt_server = "10.225.203.97";

// --- CONFIGURATION DU CLAVIER 4x4 ---
const byte ROWS = 4;
const byte COLS = 4;
char keys[ROWS][COLS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};

byte rowPins[ROWS] = {19, 18, 5, 17}; 
byte colPins[COLS] = {16, 4, 0, 2}; 

Keypad customKeypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

// --- OBJETS ---
WiFiClient espClient;
PubSubClient client(espClient);
DHT dht(DHTPIN, DHTTYPE);
Servo myServo;

// --- VARIABLES DE TEMPS ET LOGIQUE ---
unsigned long lastMqttMsg = 0;
String codeSaisi = ""; 
bool servoActif = true; // L'éolienne tourne au démarrage

void setup_wifi() {
  delay(10);
  Serial.println("\nConnexion WiFi...");
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi connected");
}

void reconnect() {
  while (!client.connected()) {
    Serial.print("Connexion MQTT...");
    String clientId = "ESP32-Client-" + String(random(0xffff), HEX);
    if (client.connect(clientId.c_str())) {
      Serial.println("connected");
    } else {
      Serial.print("failed, rc=");
      Serial.print(client.state());
      delay(2000);
    }
  }
}

void setup() {
  Serial.begin(115200);

  // Initialisation Servo
  ESP32PWM::allocateTimer(0);
  myServo.setPeriodHertz(50);
  myServo.attach(SERVO_PIN, 1000, 2000);
  
  // Sécurité pour la lecture du capteur
  pinMode(DHTPIN, INPUT_PULLUP); 
  
  dht.begin();
  setup_wifi();
  client.setServer(mqtt_server, 1883);
}

void loop() {
  if (!client.connected()) reconnect();
  client.loop();

  unsigned long now = millis();

  // --- LOGIQUE DU CLAVIER ---
  char customKey = customKeypad.getKey();
  
  if (customKey) {
    Serial.print("Touche pressée : ");
    Serial.println(customKey);
    
    codeSaisi += customKey;
    
    // On ne garde que les 4 derniers caractères tapés
    if (codeSaisi.length() > 4) {
      codeSaisi = codeSaisi.substring(codeSaisi.length() - 4);
    }

    // --- VÉRIFICATION DES CODES ---
    if (codeSaisi == "6767") {
      Serial.println("Code 6767 correct ! ARRÊT de l'éolienne.");
      servoActif = false;
      codeSaisi = ""; // On vide la mémoire après un bon code
    }
    else if (codeSaisi == "1234") { // <-- LE NOUVEAU CODE DE DÉMARRAGE EST ICI
      Serial.println("Code 1234 correct ! DÉMARRAGE de l'éolienne.");
      servoActif = true;
      codeSaisi = ""; // On vide la mémoire
    }
  }

  // --- LOGIQUE DU SERVO 360° (Optimisée) ---
  static bool etatPrecedentServo = !servoActif; // Force la mise à jour au 1er passage

  // Le signal n'est envoyé que si l'état change (évite de parasiter le DHT22)
  if (servoActif != etatPrecedentServo) {
    if (servoActif) {
      myServo.write(135); // Vitesse de rotation (ajuste si besoin)
    } else {
      myServo.write(90);  // Arrêt complet du 360°
    }
    etatPrecedentServo = servoActif;
  }

  // --- LOGIQUE MQTT (Toutes les 5 secondes) ---
  if (now - lastMqttMsg > 5000) {
    lastMqttMsg = now;
    float h = dht.readHumidity();
    float t = dht.readTemperature();

    if (!isnan(h) && !isnan(t)) {
        String payload = "{\"sensor_id\":" + String(SENSOR_ID) + 
                         ",\"temperature\":" + String(t) + 
                         ",\"humidity\":" + String(h) + "}";
                         
        client.publish(MQTT_TOPIC, payload.c_str()); 
        Serial.println("MQTT Sent: " + payload);
    } else {
        Serial.println("Erreur: Impossible de lire le capteur DHT !");
    }
  }
}
